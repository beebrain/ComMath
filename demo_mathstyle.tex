%% ══════════════════════════════════════
%% 1. นิยาม (Definition) — สีเขียวมรกต
%% ══════════════════════════════════════

\begin{definition}
\normalfont
ให้ $\mathbf{x} = [x_1, x_2, \ldots, x_n]^T$ เป็นเวกเตอร์อินพุต และ $\mathbf{w} = [w_1, w_2, \ldots, w_n]^T$ เป็นเวกเตอร์น้ำหนัก \textbf{เพอร์เซปตรอน} (Perceptron) คือ โมเดลการคำนวณที่ให้ผลลัพธ์ $y$ ตามเงื่อนไขทางคณิตศาสตร์ดังต่อไปนี้:
\[
y = \begin{cases}
1 & \text{ถ้า } \mathbf{w}^T\mathbf{x} + b > 0 \\
0 & \text{ถ้า } \mathbf{w}^T\mathbf{x} + b \leq 0
\end{cases}
\]
เมื่อ $b$ คือค่าไบแอส (Bias) ของระบบ
\end{definition}

\vspace{1em}

\begin{definition}
\normalfont
\textbf{ประพจน์} (Proposition) คือ ประโยคบอกเล่าหรือปฏิเสธ ที่มีค่าความจริงเป็น \textbf{จริง} (True) หรือ \textbf{เท็จ} (False) อย่างใดอย่างหนึ่งเท่านั้น ไม่สามารถเป็นทั้งจริงและเท็จพร้อมกันได้
\end{definition}

\vspace{2em}

%% ══════════════════════════════════════
%% 2. ทฤษฎีบท (Theorem) — สีน้ำเงินเข้ม
%% ══════════════════════════════════════

\begin{thrm}[กฎของ De Morgan]
\normalfont
สำหรับทุกเซต $A$ และ $B$ จะได้ว่า:
\begin{enumerate}[label=(\arabic*), leftmargin=2.5cm]
    \item $\overline{A \cup B} = \overline{A} \cap \overline{B}$
    \item $\overline{A \cap B} = \overline{A} \cup \overline{B}$
\end{enumerate}
\end{thrm}

\begin{proof}
พิสูจน์ข้อ (1): ให้ $x \in \overline{A \cup B}$

จะได้ $x \notin A \cup B$ ซึ่งหมายความว่า $x \notin A$ และ $x \notin B$

ดังนั้น $x \in \overline{A}$ และ $x \in \overline{B}$

สรุปได้ว่า $x \in \overline{A} \cap \overline{B}$

ดังนั้น $\overline{A \cup B} \subseteq \overline{A} \cap \overline{B}$

ในทำนองเดียวกัน สามารถพิสูจน์ได้ว่า $\overline{A} \cap \overline{B} \subseteq \overline{A \cup B}$

จึงสรุปได้ว่า $\overline{A \cup B} = \overline{A} \cap \overline{B}$
\end{proof}

\vspace{1em}

\begin{thrm}
\normalfont
สำหรับทุกประพจน์ $p$, $q$ และ $r$ จะได้ว่า
\[
(p \rightarrow q) \wedge (q \rightarrow r) \implies (p \rightarrow r)
\]
ทฤษฎีบทนี้เรียกว่า \textbf{กฎการถ่ายทอด} (Law of Syllogism)
\end{thrm}

\vspace{2em}

%% ══════════════════════════════════════
%% 3. ตัวอย่าง (Example) — สีส้ม
%% ══════════════════════════════════════

\begin{ex}
\normalfont
จงหาค่าความจริงของ $p \wedge q$ เมื่อ $p$: ``$3+2=5$'' และ $q$: ``$\pi > 3$''

\textbf{วิธีทำ.} เนื่องจาก $p$ เป็นจริง (T) และ $q$ เป็นจริง (T)

ดังนั้น $p \wedge q = T \wedge T = T$

ตอบ $p \wedge q$ มีค่าความจริงเป็น \textbf{จริง}
\end{ex}

\vspace{1em}

\begin{ex}
\normalfont
จงหาผลลัพธ์ของ $A \cup B$ เมื่อ $A = \{1, 2, 3, 4\}$ และ $B = \{3, 4, 5, 6\}$

\textbf{วิธีทำ.}
\[
A \cup B = \{1, 2, 3, 4\} \cup \{3, 4, 5, 6\} = \{1, 2, 3, 4, 5, 6\}
\]
\end{ex}

\vspace{2em}

%% ══════════════════════════════════════
%% 4. บทแทรก (Lemma) — สีม่วง

\begin{prop}
\normalfont
ถ้า $A \subseteq B$ และ $B \subseteq C$ แล้ว $A \subseteq C$
\end{prop}

\vspace{2em}

%% ══════════════════════════════════════
%% 5. บทสรุป (Corollary) — สีเขียวน้ำทะเล
%% ══════════════════════════════════════

\begin{cor}
\normalfont
ถ้า $A = B$ แล้ว $\overline{A} = \overline{B}$
\end{cor}

\begin{rem}
\normalfont
ประโยคคำถาม ประโยคอุทาน และประโยคคำสั่ง ไม่ถือเป็นประพจน์ เนื่องจากไม่สามารถตัดสินค่าความจริงได้ ตัวอย่างเช่น ``$x + 2 = 5$'' ไม่เป็นประพจน์ เพราะค่าความจริงขึ้นอยู่กับค่าของ $x$
\end{rem}
